Identifiers and keywords¶

identifier  ::=  unquoted_identifier | quoted_identifier
                  unquoted_identifier ::=  re('[a-zA-Z][a-zA-Z0-9_]*')
                  quoted_identifier   ::=  '"' (any character where " can appear if doubled)+ '"'
